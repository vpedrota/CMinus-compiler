/* programa para calcular o fatorial de um numero */

int factorial(int n) {
    int res;
    int i;

    res = 1;
    i = 1;
    
    while (i <= n) {
        res = res * i;
        i = i + 1;
    }

    return res;
}

void main(void) {

    int num;
    int result;
  
    num = 4;
    result = factorial(num);
    output(result);
    
}
