/* programa para calcular a potencia de um numero */

int base;
int exponent;
int result;

int power(int b, int e) {
    int res;
    int i;

    res = 1;
    i = 0;
    
    if (e < 0) {
        return 0;
    }

    while (i < e) {
        res = res * b;
        i = i + 1;
    }

    return res;
}

void main(void) {
    int i;
    base = input();

    exponent = input();

    result = power(base, exponent);
    output(result);
}
